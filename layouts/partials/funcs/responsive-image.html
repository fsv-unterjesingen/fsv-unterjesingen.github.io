$resolvedURL = $imageURL | relURL -}} {{- end -}} {{- end -}} {{- end -}} {{- if not $fallback -}} {{- $fallback = $resolvedURL -}} {{- end -}} {{- return (dict "url" $resolvedURL "fallback" $fallback
